## Unit Testing Principles

Unit tests focus on testing individual functions in isolation:

**Unit Test Scope:** Test pure functions, utilities, and business logic without external dependencies like databases, APIs, or UI components.

<Tab value="Test Structure">
**Organize Tests by Function:**
```typescript
// src/lib/__tests__/crypto.test.ts
import { describe, it, expect } from 'vitest'
import { hashPassword, verifyPassword, generateApiKey } from '../crypto'

describe('Crypto Utilities', () => {
  describe('hashPassword', () => {
    it('should hash password with bcrypt format', async () => {
      const password = 'TestPassword123!'
      const hash = await hashPassword(password)

      expect(hash).not.toBe(password)
      expect(hash).toMatch(/^\$2[aby]?\$/) // bcrypt format
      expect(hash.length).toBeGreaterThan(50)
    })

    it('should generate different hashes for same password', async () => {
      const password = 'TestPassword123!'
      const hash1 = await hashPassword(password)
      const hash2 = await hashPassword(password)

      expect(hash1).not.toBe(hash2) // Salt should make them different
    })
  })

  describe('verifyPassword', () => {
    it('should verify correct password', async () => {
      const password = 'TestPassword123!'
      const hash = await hashPassword(password)

      const isValid = await verifyPassword(password, hash)
      expect(isValid).toBe(true)
    })

    it('should reject incorrect password', async () => {
      const password = 'TestPassword123!'
      const wrongPassword = 'WrongPassword123!'
      const hash = await hashPassword(password)

      const isValid = await verifyPassword(wrongPassword, hash)
      expect(isValid).toBe(false)
    })
  })
})
```

**Key Principles:**
- **One Function Per Describe Block**: Group related tests
- **Descriptive Test Names**: Clearly state what is being tested
- **Arrange-Act-Assert**: Setup, execute, verify pattern
- **Test Edge Cases**: Include boundary conditions and error cases
</Tab>

<Tab value="Best Practices">
**Testing Best Practices:**
```typescript
// ✅ Good: Clear, focused test
describe('calculateTax', () => {
  it('should calculate correct tax for standard rate', () => {
    const result = calculateTax(100, 0.2)
    expect(result).toBe(20)
  })

  it('should handle zero amount', () => {
    const result = calculateTax(0, 0.2)
    expect(result).toBe(0)
  })

  it('should handle zero rate', () => {
    const result = calculateTax(100, 0)
    expect(result).toBe(0)
  })
})

// ❌ Bad: Vague, multiple concerns
describe('tax stuff', () => {
  it('should work', () => {
    expect(calculateTax(100, 0.2)).toBeTruthy()
    expect(validateTaxRate(0.5)).toBe(true)
    expect(formatTaxAmount(20)).toContain('$')
  })
})
```

**Test Isolation:**
```typescript
// ✅ Good: Each test is independent
describe('UserValidator', () => {
  it('should validate email format', () => {
    expect(validateEmail('test@example.com')).toBe(true)
    expect(validateEmail('invalid-email')).toBe(false)
  })

  it('should validate password strength', () => {
    expect(validatePassword('weak')).toBe(false)
    expect(validatePassword('StrongPass123!')).toBe(true)
  })
})

// Each test can run independently in any order
```

**Error Testing:**
```typescript
describe('Error Handling', () => {
  it('should throw error for invalid input', () => {

    }).toThrow('Invalid date format')

    expect(() => {
      validateAge(-5)
```

<Tab value="Performance Tips">
**Optimizing Unit Tests:**
```typescript
// ✅ Good: Use test.each for multiple similar tests
describe('formatCurrency', () => {
  test.each([
    [1999, '$19.99'],
    [0, '$0.00'],
    [12345, '$123.45'],
    [5, '$0.05'],
  ])('should format %i cents as %s', (cents, expected) => {
    expect(formatCurrency(cents)).toBe(expected)
  })
})

// ✅ Good: Expensive setup in beforeAll (not beforeEach)
describe('ComplexCalculation', () => {
  let calculator: Calculator

  beforeAll(() => {
    // Expensive setup only once
    calculator = new Calculator(complexConfiguration)
  })

  it('should calculate result A', () => {
    expect(calculator.calculate('A')).toBe(expectedA)
  })

  it('should calculate result B', () => {
    expect(calculator.calculate('B')).toBe(expectedB)
  })
})

// ✅ Good: Avoid unnecessary async/await
describe('SyncFunction', () => {
  it('should return result immediately', () => {
    // No need for async/await if function is sync
    const result = syncFunction(input)
    expect(result).toBe(expected)
  })
})
```

```typescript
    expect(addDays(endOfMonth, 1)).toEqual(new Date('2024-02-01'))
```

## Best Practices Summary

<Callout type="success">
**Unit Testing Excellence:**
- **Test Pure Functions**: Focus on functions without side effects
- **Clear Test Names**: Describe exactly what is being tested
- **Fast Execution**: Unit tests should run quickly in isolation
- **Edge Cases**: Test boundary conditions and error scenarios
</Callout>

### **Key Principles:**

1. **Isolation**: Each test should be independent and not rely on external state
2. **Deterministic**: Tests should produce the same result every time
3. **Fast**: Unit tests should execute quickly (< 10ms per test)
4. **Focused**: One assertion per test, testing one specific behavior
5. **Readable**: Tests should serve as documentation of the function's behavior

### **Common Patterns:**

```typescript
// ✅ Good unit test structure
describe('FunctionName', () => {
  describe('when input is valid', () => {
    it('should return expected result', () => {
      const result = functionName(validInput)
      expect(result).toBe(expectedOutput)
    })
  })

  describe('when input is invalid', () => {
    it('should throw ValidationError', () => {
      expect(() => functionName(invalidInput)).toThrow(ValidationError)
    })
  })
})
```
