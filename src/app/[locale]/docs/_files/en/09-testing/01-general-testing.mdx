// Wait for condition with timeout
export const waitForCondition = async (
  condition: () => boolean | Promise<boolean>,
  timeout = 5000,
  interval = 100
): Promise<void> => {
  const startTime = Date.now()

  while (Date.now() - startTime < timeout) {
    if (await condition()) {
      return
    }
    await new Promise(resolve => setTimeout(resolve, interval))
  }

  throw new Error(`Condition not met within ${timeout}ms`)
}

// Wait for mock to be called
export const waitForMockCall = async (
  mockFn: any,
  callCount = 1,
  timeout = 5000
) => {
  await waitForCondition(
    () => mockFn.mock.calls.length >= callCount,
    timeout
  )
}

// Retry with backoff
export const retryWithBackoff = async <T>(
  fn: () => Promise<T>,
  maxRetries = 3,
  baseDelay = 100
): Promise<T> => {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn()
    } catch (error) {
      if (i === maxRetries - 1) throw error

      const delay = baseDelay * Math.pow(2, i)
      await new Promise(resolve => setTimeout(resolve, delay))
    }
  }
  throw new Error('Max retries exceeded')
}

// Usage examples
describe('Async Operations', () => {
  it('should process async task', async () => {
    const processTask = vi.fn()

    // Start async process
    someAsyncService.process()

    // Wait for processing to complete
    await waitForMockCall(processTask)

    expect(processTask).toHaveBeenCalled()
  })

  it('should handle flaky operations', async () => {
    const flakyOperation = vi.fn()
      .mockRejectedValueOnce(new Error('Temporary failure'))
      .mockResolvedValue({ success: true })

    const result = await retryWithBackoff(() => flakyOperation())

import { vi } from 'vitest'

// Create typed mock
export const createMockFunction = <T extends (...args: any[]) => any>(
  implementation?: T
) => {
  const mock = vi.fn(implementation)
  return mock as vi.MockedFunction<T>
}

// Mock implementation builder
export class MockBuilder<T> {
  private mocks: Partial<T> = {}

  method<K extends keyof T>(key: K, implementation: T[K]): MockBuilder<T> {
    this.mocks[key] = vi.fn(implementation) as T[K]
    return this
  }

  property<K extends keyof T>(key: K, value: T[K]): MockBuilder<T> {
    this.mocks[key] = value
    return this
  }

  build(): T {
    return this.mocks as T
  }
}

// Repository mock factory
export const createRepositoryMock = <T>(defaultReturns: Partial<Record<keyof T, any>> = {}) => {
  const mock: any = {}

  // Common repository methods

  methods.forEach(method => {
    mock[method] = vi.fn().mockResolvedValue(defaultReturns[method] || null)
  })

  return mock as T

## Best Practices

<Callout type="success">
**General Testing Principles:**
- **Consistency**: Use factories and utilities for consistent test data
- **Isolation**: Each test should be independent and repeatable
- **Clarity**: Test names and structure should be self-documenting
- **Performance**: Mock external dependencies to keep tests fast
</Callout>

### **Key Recommendations:**

1. **Use Factories**: Create reusable data factories for consistent test data
2. **Mock External Services**: Keep tests focused on your code, not third-party services
3. **Clean Up**: Always clean up after tests to prevent pollution
4. **Type Safety**: Leverage TypeScript for better test reliability
5. **Shared Utilities**: Create helper functions for common testing patterns
